 # ANSI Escape Code Structure:
 # \033[<style>;<foreground>;<background>m<text>\033[0m
 #
 # Style codes: 0=normal, 1=bold, 4=underline
 # Foreground (text) colors: 30-37 (add 60 for bright versions: 90-97)
 # Reset: \033[0m

 class Colors:
     # Bright colors for better visibility
     RED = '\033[91m'      # Errors
     GREEN = '\033[92m'    # Success
     BLUE = '\033[94m'     # Info
     YELLOW = '\033[93m'   # Warnings
     CYAN = '\033[96m'     # Headings/borders
     MAGENTA = '\033[95m'  # Special items
     WHITE = '\033[97m'
     GRAY = '\033[90m'     # Subtle text

     # Styles
     BOLD = '\033[1m'
     UNDERLINE = '\033[4m'

     # Reset
     RESET = '\033[0m'

 def colorize(text, color):
     """Wrap text in a color code and reset."""
     return f"{color}{text}{Colors.RESET}"

 def success(text):
     """Return green text for success messages."""
     return colorize(text, Colors.GREEN)

 def error(text):
     """Return red text for error messages."""
     return colorize(text, Colors.RED)

 def info(text):
     """Return blue text for info messages."""
     return colorize(text, Colors.BLUE)

 def warning(text):
     """Return yellow text for warnings."""
     return colorize(text, Colors.YELLOW)

 def heading(text):
     """Return cyan bold text for headings."""
     return colorize(text, Colors.CYAN + Colors.BOLD)

 # Box drawing characters (Unicode)
 class BoxChars:
     TOP_LEFT = '┌'
     TOP_RIGHT = '┐'
     BOTTOM_LEFT = '└'
     BOTTOM_RIGHT = '┘'
     HORIZONTAL = '─'
     VERTICAL = '│'
     T_DOWN = '┬'      # Top T-junction
     T_UP = '┴'        # Bottom T-junction
     T_RIGHT = '├'     # Left T-junction
     T_LEFT = '┤'      # Right T-junction
     CROSS = '┼'       # Cross junction

 def draw_box(title, content_lines, width=60):
     """
     Draw a box around content.

     Args:
         title (str): Title for the box (appears at top)
         content_lines (list): List of strings, each is a line inside the box
         width (int): Total width of the box

     Returns:
         str: The complete box as a string
     """
     lines = []

     # Top border with title
     if title:
         title_formatted = f" {title} "
         padding_left = (width - len(title_formatted) - 2) // 2
         padding_right = width - len(title_formatted) - padding_left - 2
         top_line = (BoxChars.TOP_LEFT +
                    BoxChars.HORIZONTAL * padding_left +
                    title_formatted +
                    BoxChars.HORIZONTAL * padding_right +
                    BoxChars.TOP_RIGHT)
     else:
         top_line = BoxChars.TOP_LEFT + BoxChars.HORIZONTAL * (width - 2) + BoxChars.TOP_RIGHT

     lines.append(colorize(top_line, Colors.CYAN))

     # Content lines
     for line in content_lines:
         padded = line.ljust(width - 4)  # -4 for borders and spaces
         content_line = f"{colorize(BoxChars.VERTICAL, Colors.CYAN)} {padded} {colorize(BoxChars.VERTICAL, Colors.CYAN)}"
         lines.append(content_line)

     # Bottom border
     bottom_line = BoxChars.BOTTOM_LEFT + BoxChars.HORIZONTAL * (width - 2) + BoxChars.BOTTOM_RIGHT
     lines.append(colorize(bottom_line, Colors.CYAN))

     return '\n'.join(lines)

 def draw_table(headers, rows, widths=None):
     """
     Draw a formatted table.

     Args:
         headers (list): List of header strings
         rows (list): List of row tuples/lists
         widths (list): Optional list of column widths

     Returns:
         str: The formatted table
     """
     if not widths:
         # Auto-calculate widths based on content
         widths = []
         for i, header in enumerate(headers):
             max_width = len(header)
             for row in rows:
                 if i < len(row):
                     max_width = max(max_width, len(str(row[i])))
             widths.append(max_width + 2)  # +2 for padding

     lines = []

     # Top border
     top = BoxChars.TOP_LEFT
     for i, width in enumerate(widths):
         top += BoxChars.HORIZONTAL * width
         if i < len(widths) - 1:
             top += BoxChars.T_DOWN
     top += BoxChars.TOP_RIGHT
     lines.append(colorize(top, Colors.CYAN))

     # Header row
     header_row = colorize(BoxChars.VERTICAL, Colors.CYAN)
     for i, (header, width) in enumerate(zip(headers, widths)):
         header_row += colorize(f" {header.ljust(width - 1)}", Colors.CYAN + Colors.BOLD)
         header_row += colorize(BoxChars.VERTICAL, Colors.CYAN)
     lines.append(header_row)

     # Header separator
     separator = colorize(BoxChars.T_RIGHT, Colors.CYAN)
     for i, width in enumerate(widths):
         separator += colorize(BoxChars.HORIZONTAL * width, Colors.CYAN)
         if i < len(widths) - 1:
             separator += colorize(BoxChars.CROSS, Colors.CYAN)
     separator += colorize(BoxChars.T_LEFT, Colors.CYAN)
     lines.append(separator)

     # Data rows
     for row in rows:
         row_line = colorize(BoxChars.VERTICAL, Colors.CYAN)
         for i, (cell, width) in enumerate(zip(row, widths)):
             row_line += f" {str(cell).ljust(width - 1)}"
             row_line += colorize(BoxChars.VERTICAL, Colors.CYAN)
         lines.append(row_line)

     # Bottom border
     bottom = BoxChars.BOTTOM_LEFT
     for i, width in enumerate(widths):
         bottom += BoxChars.HORIZONTAL * width
         if i < len(widths) - 1:
             bottom += BoxChars.T_UP
     bottom += BoxChars.BOTTOM_RIGHT
     lines.append(colorize(bottom, Colors.CYAN))

     return '\n'.join(lines)

 def horizontal_rule(width=60, char='─', color=Colors.CYAN):
     """Create a horizontal line separator."""
     return colorize(char * width, color)
